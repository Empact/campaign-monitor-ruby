    def lists
      response = @cm_client.Client_GetLists("ClientID" => @id)
      unless response["Code"].to_i != 0 
        response["List"].collect{|l| List.new(l["ListID"].to_i, l["Name"])}
      else
        raise response["Code"] + " - " + response["Message"]
      end
    end
